This guide explains how to handle errors effectively when using the SD2 Mod API.

## Error Response Format

All API errors follow a consistent format:

```json
{
  "status": "error",
  "error": {
    "message": "Human-readable error message",
    "code": "ERROR_CODE",
    "fields": {
      "field_name": ["Error message for this field"]
    }
  }
}
```

### Basic Error Handling

<CodeGroup>

```typescript TypeScript
interface ApiError {
  status: 'error'
  error: {
    message: string
    code?: string
    fields?: Record<string, string[]>
  }
}

async function makeApiRequest<T>(
  url: string,
  options: RequestInit = {}
): Promise<T> {
  try {
    const response = await fetch(url, {
      ...options,
      headers: {
        'Content-Type': 'application/json',
        'x-api-key': process.env.API_KEY,
        ...options.headers,
      },
    })

    if (!response.ok) {
      const error = (await response.json()) as ApiError
      throw new Error(error.error.message)
    }

    return response.json()
  } catch (error) {
    console.error('API request failed:', error)
    throw error
  }
}

// Example usage
try {
  const order = await makeApiRequest('https://api.orders.sdigital2.com/api/orders/123')
  console.log('Order:', order)
} catch (error) {
  console.error('Failed to fetch order:', error)
}
```

```python Python
from typing import TypedDict, Optional, Dict, List, Any
import requests

class ApiError(TypedDict):
    status: str
    error: Dict[str, Any]

def make_api_request(
    url: str,
    method: str = 'GET',
    data: Optional[dict] = None
) -> Any:
    try:
        response = requests.request(
            method,
            url,
            headers={
                'Content-Type': 'application/json',
                'x-api-key': os.getenv('API_KEY')
            },
            json=data
        )

        response.raise_for_status()
        return response.json()
    except requests.exceptions.RequestException as e:
        if e.response is not None:
            error = e.response.json()
            raise Exception(error['error']['message']) from e
        raise

# Example usage
try:
    order = make_api_request('https://api.orders.sdigital2.com/api/orders/123')
    print('Order:', order)
except Exception as error:
    print('Failed to fetch order:', error)
```

### Retry Logic

Implement exponential backoff for retrying failed requests:

<CodeGroup>

```typescript TypeScript
async function makeApiRequestWithRetry<T>(
  url: string,
  options: RequestInit = {},
  maxRetries = 3
): Promise<T> {
  let lastError: Error | null = null

  for (let attempt = 1; attempt <= maxRetries; attempt++) {
    try {
      return await makeApiRequest<T>(url, options)
    } catch (error) {
      lastError = error as Error

      // Don't retry certain errors
      if (error instanceof Response) {
        if ([401, 403, 404].includes(error.status)) {
          throw error
        }
      }

      if (attempt === maxRetries) break

      // Exponential backoff
      const delay = Math.pow(2, attempt) * 1000
      await new Promise(resolve => setTimeout(resolve, delay))
    }
  }

  throw new Error(`Failed after ${maxRetries} attempts: ${lastError?.message}`)
}

// Example usage
try {
  const order = await makeApiRequestWithRetry(
    'https://api.orders.sdigital2.com/api/orders/123'
  )
  console.log('Order:', order)
} catch (error) {
  console.error('Failed to fetch order:', error)
}
```

```python Python
import time
from typing import Any, Optional

def make_api_request_with_retry(
    url: str,
    method: str = 'GET',
    data: Optional[dict] = None,
    max_retries: int = 3
) -> Any:
    last_error = None

    for attempt in range(1, max_retries + 1):
        try:
            return make_api_request(url, method, data)
        except requests.exceptions.RequestException as error:
            last_error = error

            # Don't retry certain errors
            if error.response is not None:
                if error.response.status_code in [401, 403, 404]:
                    raise

            if attempt == max_retries:
                break

            # Exponential backoff
            delay = 2 ** attempt
            time.sleep(delay)

    raise Exception(
        f"Failed after {max_retries} attempts: {str(last_error)}"
    )

# Example usage
try:
    order = make_api_request_with_retry(
        'https://api.orders.sdigital2.com/api/orders/123'
    )
    print('Order:', order)
except Exception as error:
    print('Failed to fetch order:', error)
```

### Rate Limit Handling

Handle rate limit errors with proper backoff:

<CodeGroup>

```typescript TypeScript
async function makeApiRequestWithRateLimit<T>(
  url: string,
  options: RequestInit = {}
): Promise<T> {
  const response = await fetch(url, {
    ...options,
    headers: {
      'Content-Type': 'application/json',
      'x-api-key': process.env.API_KEY,
      ...options.headers,
    },
  })

  if (response.status === 429) {
    const retryAfter = response.headers.get('retry-after')
    const delayMs = retryAfter ? parseInt(retryAfter) * 1000 : 5000

    console.warn(
      `Rate limit exceeded. Retrying after ${delayMs / 1000} seconds`
    )

    await new Promise(resolve => setTimeout(resolve, delayMs))
    return makeApiRequestWithRateLimit<T>(url, options)
  }

  if (!response.ok) {
    const error = await response.json()
    throw new Error(error.error.message)
  }

  return response.json()
}

// Example usage
try {
  const orders = await makeApiRequestWithRateLimit(
    'https://api.orders.sdigital2.com/api/orders'
  )
  console.log('Orders:', orders)
} catch (error) {
  console.error('Failed to fetch orders:', error)
}
```

```python Python
def make_api_request_with_rate_limit(
    url: str,
    method: str = 'GET',
    data: Optional[dict] = None
) -> Any:

            method,
            url,
            headers={
                'Content-Type': 'application/json',

        'https://api.orders.sdigital2.com/api/orders'
```

### Field Validation Errors

Handle validation errors for specific fields:

<CodeGroup>

```typescript TypeScript
interface ValidationError {
  status: 'error'
  error: {
    message: string
    fields: Record<string, string[]>
  }
}

function isValidationError(error: unknown): error is ValidationError {
  return (
    typeof error === 'object' &&
    error !== null &&
    'status' in error &&
    error.status === 'error' &&
    'error' in error &&
    typeof error.error === 'object' &&
    error.error !== null &&
    'fields' in error.error
  )
}

async function createOrder(orderData: OrderCreate) {
  try {
    const response = await fetch('https://api.orders.sdigital2.com/api/orders', {
      method: 'POST',
      headers: {
        'Content-Type': 'application/json',
        'x-api-key': process.env.API_KEY,
      },
      body: JSON.stringify(orderData),
    })

    if (!response.ok) {
      const error = await response.json()

      if (isValidationError(error)) {
        console.error('Validation errors:', error.error.fields)
        // Handle specific field errors
        if (error.error.fields['from.amount']) {
          console.error('Invalid amount:', error.error.fields['from.amount'])
        }
      }

      throw new Error(error.error.message)
    }

    return response.json()
  } catch (error) {
    console.error('Failed to create order:', error)
    throw error
  }
}
```

```python Python
from typing import TypedDict, Dict, List

class ValidationError(TypedDict):
    status: str
    error: Dict[str, Dict[str, List[str]]]

def is_validation_error(error: Any) -> bool:
    return (
        isinstance(error, dict) and
        error.get('status') == 'error' and
        isinstance(error.get('error'), dict) and
        'fields' in error['error']
    )

def create_order(order_data: dict):
    try:
        response = requests.post(
            'https://api.orders.sdigital2.com/api/orders',
            headers={
                'Content-Type': 'application/json',
                'x-api-key': os.getenv('API_KEY')
            },
            json=order_data
        )

        if not response.ok:
            error = response.json()

            if is_validation_error(error):
                print('Validation errors:', error['error']['fields'])
                # Handle specific field errors
                if 'from.amount' in error['error']['fields']:
                    print(
                        'Invalid amount:',
                        error['error']['fields']['from.amount']
                    )

            raise Exception(error['error']['message'])

        return response.json()
    except Exception as error:
        print('Failed to create order:', error)
        raise
```

## Best Practices

1. **Always Use Try-Catch**

   - Wrap API calls in try-catch blocks
   - Handle specific error types
   - Provide meaningful error messages

2. **Implement Retries**

   - Use exponential backoff
   - Set maximum retry attempts
   - Skip retries for certain errors

3. **Validate Input**

   - Check data before sending
   - Handle validation errors gracefully
   - Provide clear error messages

4. **Log Errors**
   - Include relevant context
   - Don't log sensitive data
   - Use proper log levels

## Rate Limits

- 1000 requests per hour for most endpoints
- Retries don't count against limits if proper backoff is used
- Contact support for increased limits

<Note>
  Use exponential backoff when retrying requests to avoid overwhelming the API.
</Note>
